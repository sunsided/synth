1//! Trapezoidal state-variable filter (TPT SVF) with pre-drive saturation.
2//!
3//! Reference: Andrew Simper, "Solving the Continuous SVF Equations Using
4//! Trapezoidal Integration and Equivalent Circuits" (2013).
5
6use crate::params::FilterMode;
7use std::f32::consts::PI;
8
9/// Trapezoidal-integrated state-variable filter (Cytomic/Simper TPT SVF).
10///
11/// Unconditionally stable for any sample rate and cutoff.  Provides LP, BP,
12/// and HP simultaneously.  A subtle `tanh` saturation is applied to the
13/// band-pass state for SID-style resonance character.
14///
15/// Pre-drive soft-clipping of the input is also supported (maps to the SID
16/// "drive" character control).
17pub struct SvFilter {
18    ic1eq: f32, // first integrator state
19    ic2eq: f32, // second integrator state
20}
21
22impl Default for SvFilter {
23    /// Create a filter with zeroed integrator states.
24    fn default() -> Self {
25        Self {
26            ic1eq: 0.0,
27            ic2eq: 0.0,
28        }
29    }
30}
31
32impl SvFilter {
33    /// Zero both integrator states (use after a voice panic or hard reset).
34    pub fn reset(&mut self) {
35        self.ic1eq = 0.0;
36        self.ic2eq = 0.0;
37    }
38
39    /// Process a single sample.
40    ///
41    /// * `input`      – input sample (-1..1 range expected)
42    /// * `mode`       – LP / BP / HP
43    /// * `cutoff`     – cutoff frequency in Hz
44    /// * `resonance`  – 0.0 (no resonance) .. 0.99 (very high Q)
45    /// * `drive`      – 0.0 .. 1.0, pre-filter saturation
46    /// * `sample_rate`– audio sample rate in Hz
47    pub fn process(
48        &mut self,
49        input: f32,
50        mode: FilterMode,
51        cutoff: f32,
52        resonance: f32,
53        drive: f32,
54        sample_rate: f32,
55    ) -> f32 {
56        // Pre-drive soft clip (SID "grit")
57        let x = if drive > 0.001 {
58            let gain = 1.0 + drive * 4.0;
59            let x = input * gain;
60            // Soft clip: x / sqrt(1 + x²)  (keeps odd harmonics, no hard edge)
61            let clipped = x / (1.0 + x * x).sqrt();
62            // When gain·input overflows f32 to ±Inf, x²→Inf and we get Inf/Inf=NaN.
63            // The analytic limit as |x|→∞ is ±1; recover with copysign.
64            if clipped.is_finite() {
65                clipped
66            } else {
67                std::hint::cold_path();
68                1.0_f32.copysign(x)
69            }
70        } else {
71            input
72        };
73
74        // Filter coefficient calculation
75        // Clamp the normalised cutoff ratio to just below 0.5 before computing the
76        // bilinear warp coefficient `g`.  Two edge cases are handled here:
77        //   1. sample_rate < ~40 Hz: Nyquist would fall below the 20 Hz minimum,
78        //      so we raise the Nyquist floor to 20 Hz to keep clamp(min, max) valid.
79        //   2. Very small sample_rate: even with fc clamped, PI*fc/sr can overflow
80        //      f32 to Inf, causing tan(Inf) = NaN.  Clamping fc_norm prevents this.
81        let nyquist = (sample_rate * 0.499).max(20.0);
82        let fc = cutoff.clamp(20.0, nyquist);
83        // fc_norm ∈ (0, 0.499] ensures PI*fc_norm < PI/2, keeping tan finite.
84        let fc_norm = (fc / sample_rate).min(0.499_f32);
85        let g = (PI * fc_norm).tan();
86        // k = 2 - 2*resonance maps resonance 0→k=2 (Q=0.5) to 0.99→k=0.02 (Q≈50)
87        let k = (2.0 - 1.98 * resonance.clamp(0.0, 0.999)).max(0.01);
88
89        let a1 = 1.0 / (1.0 + g * (g + k));
90        let a2 = g * a1;
91        let a3 = g * a2;
92
93        let v3 = x - self.ic2eq;
94        let v1 = a1 * self.ic1eq + a2 * v3;
95        let v2 = self.ic2eq + a2 * self.ic1eq + a3 * v3;
96
97        // Update integrators
98        self.ic1eq = 2.0 * v1 - self.ic1eq;
99        self.ic2eq = 2.0 * v2 - self.ic2eq;
100
101        // Clamp integrators to prevent denormals / instability under extreme params
102        self.ic1eq = clamp_denormal(self.ic1eq);
103        self.ic2eq = clamp_denormal(self.ic2eq);
104
105        let out = match mode {
106            FilterMode::LowPass => v2,
107            // SID character: subtle tanh saturation on band-pass path
108            FilterMode::BandPass => fast_tanh(v1 * (1.0 + resonance * 0.5)),
109            FilterMode::HighPass => x - k * v1 - v2,
110        };
111
112        // Safety: intermediate overflow (e.g. near-f32::MAX input without soft-clip)
113        // can produce a non-finite output on the overflow sample even though
114        // clamp_denormal above ensures state recovery on the next sample.
115        // Return 0.0 rather than propagating Inf/NaN to the caller.
116        if out.is_finite() {
117            out
118        } else {
119            std::hint::cold_path();
120            0.0
121        }
122    }
123}
124
125/// Flush denormals to zero and reset any non-finite values to zero.
126///
127/// Two situations are handled:
128/// * **Denormals** (`|x| < 1e-15`): flushed to zero to avoid CPU performance
129///   degradation from subnormal float arithmetic.
130/// * **Non-finite values**: reset to zero so that an Inf/NaN integrator state
131///   (e.g., from an extremely large input sample) does not propagate to all
132///   subsequent outputs.  The reset causes a brief transient artefact but
133///   restores filter operation immediately on the next sample.
134#[inline]
135fn clamp_denormal(x: f32) -> f32 {
136    if !x.is_finite() || x.abs() < 1e-15 {
137        std::hint::cold_path();
138        0.0
139    } else {
140        x
141    }
142}
143
144/// Fast tanh approximation (Padé 5/4) – accurate to ±0.5% for |x| < 4.
145///
146/// Returns exactly `±1.0` for `|x| >= 4` so the polynomial intermediate values
147/// stay within `f32` range regardless of input magnitude.  Without this guard,
148/// `x²` overflows to `f32::INFINITY` for large inputs, causing `Inf / Inf = NaN`
149/// to escape the final clamp.
150#[inline]
151fn fast_tanh(x: f32) -> f32 {
152    // tanh(±4) ≈ ±0.9993; clamping here is within the stated ±0.5% accuracy
153    // window and prevents intermediate overflow for any finite input.
154    if x >= 4.0 {
155        std::hint::cold_path();
156        return 1.0;
157    }
158    if x <= -4.0 {
159        std::hint::cold_path();
160        return -1.0;
161    }
162    let x2 = x * x;
163    let n = x * (135.0 + x2 * (17.0 + x2));
164    let d = 135.0 + x2 * (45.0 + x2 * 9.0);
165    (n / d).clamp(-1.0, 1.0)
166}

168#[cfg(test)]
169mod tests {
170    use super::*;
171
172    #[allow(clippy::cast_precision_loss)] // small sample count; precision loss negligible
173    fn rms(samples: &[f32]) -> f32 {
174        let sum_sq: f32 = samples.iter().map(|&s| s * s).sum();
175        (sum_sq / samples.len() as f32).sqrt()
176    }
177
178    #[test]
179    #[allow(clippy::cast_precision_loss)] // test signal generation; index fits well within f32
180    fn lp_attenuates_high_freq() {
181        // 1 kHz tone, LP at 200 Hz → should be significantly attenuated
182        let mut filt = SvFilter::default();
183        let sr = 44100.0_f32;
184        let tone: Vec<f32> = (0..4096)
185            .map(|i| (2.0 * std::f32::consts::PI * 1000.0 * i as f32 / sr).sin())
186            .collect();
187        let filtered: Vec<f32> = tone
188            .iter()
189            .map(|&s| filt.process(s, FilterMode::LowPass, 200.0, 0.1, 0.0, sr))
190            .collect();
191        let ratio = rms(&filtered) / rms(&tone[2048..]); // skip transient
192        assert!(ratio < 0.5, "LP did not attenuate: ratio={ratio}");
193    }
194
195    #[test]
196    #[allow(clippy::cast_precision_loss)] // test signal generation; index fits well within f32
197    fn hp_attenuates_low_freq() {
198        // 50 Hz tone, HP at 500 Hz → should be significantly attenuated
199        let mut filt = SvFilter::default();
200        let sr = 44100.0_f32;
201        let tone: Vec<f32> = (0..8192)
202            .map(|i| (2.0 * std::f32::consts::PI * 50.0 * i as f32 / sr).sin())
203            .collect();
204        let filtered: Vec<f32> = tone
205            .iter()
206            .map(|&s| filt.process(s, FilterMode::HighPass, 500.0, 0.1, 0.0, sr))
207            .collect();
208        let ratio = rms(&filtered[4096..]) / rms(&tone[4096..]);
209        assert!(ratio < 0.5, "HP did not attenuate: ratio={ratio}");
210    }

212    #[test]
213    fn no_nan_under_extreme_params() {
214        let mut filt = SvFilter::default();
215        let input = 1.0_f32;
216        for &res in &[0.0_f32, 0.5, 0.95, 0.999] {
217            for &drive in &[0.0_f32, 0.5, 1.0] {
218                let out = filt.process(input, FilterMode::LowPass, 100.0, res, drive, 44100.0);
219                assert!(out.is_finite(), "NaN/Inf at res={res}, drive={drive}");
220            }
221        }
222    }
