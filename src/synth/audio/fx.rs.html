1//! Freeverb-inspired Schroeder reverb: 4 parallel comb filters → 2 serial allpass sections.
2//!
3//! All delay buffers are allocated once at construction; the audio callback
4//! performs no heap allocation.
5
6/// Comb filter delay lengths tuned for 44100 Hz (Freeverb defaults, slightly reduced).
7const COMB_DELAYS: [usize; 4] = [1116, 1188, 1277, 1356];
8
9/// Allpass filter delay lengths tuned for 44100 Hz.
10const ALLPASS_DELAYS: [usize; 2] = [556, 441];
11
12/// Feedback comb filter with first-order low-pass damping (Schroeder/Freeverb style).
13///
14/// The LP on the feedback path simulates high-frequency air absorption in a room.
15struct CombFilter {
16    /// Delay line ring buffer.
17    buf: Vec<f32>,
18    /// Current write/read index into `buf`.
19    idx: usize,
20    /// Feedback gain (controls room decay time).
21    feedback: f32,
22    /// Damping coefficient for the LP on the feedback path.
23    damp: f32,
24    /// State variable for the one-pole LP.
25    damp_state: f32,
26}
27
28impl CombFilter {
29    /// Construct a comb filter with the given delay length (in samples).
30    fn new(delay: usize) -> Self {
31        Self {
32            buf: vec![0.0; delay],
33            idx: 0,
34            feedback: 0.84,
35            damp: 0.5,
36            damp_state: 0.0,
37        }
38    }
39
40    /// Update feedback gain and damping without reallocating the delay line.
41    fn set_params(&mut self, feedback: f32, damp: f32) {
42        self.feedback = feedback;
43        self.damp = damp.clamp(0.0, 0.999);
44    }
45
46    /// Process one sample through the comb filter.
47    fn process(&mut self, input: f32) -> f32 {
48        let out = self.buf[self.idx];
49        // First-order low-pass on feedback path (simulates air absorption)
50        self.damp_state = out * (1.0 - self.damp) + self.damp_state * self.damp;
51        self.buf[self.idx] = input + self.damp_state * self.feedback;
52        self.idx = (self.idx + 1) % self.buf.len();
53        out
54    }
55}
56
57/// Schroeder allpass section (fixed gain of ±0.5).
58struct AllpassFilter {
59    /// Delay line ring buffer.
60    buf: Vec<f32>,
61    /// Current write/read index into `buf`.
62    idx: usize,
63}
64
65impl AllpassFilter {
66    /// Construct an allpass filter with the given delay length (in samples).
67    fn new(delay: usize) -> Self {
68        Self {
69            buf: vec![0.0; delay],
70            idx: 0,
71        }
72    }
73
74    /// Process one sample through the allpass section.
75    fn process(&mut self, input: f32) -> f32 {
76        let buf_out = self.buf[self.idx];
77        let out = buf_out - input;
78        self.buf[self.idx] = input + buf_out * 0.5;
79        self.idx = (self.idx + 1) % self.buf.len();
80        out
81    }
82}

84/// Freeverb-style plate reverb unit.
85///
86/// Four parallel comb filters feed into two serial allpass sections.
87/// Room size and damping are adjustable at runtime via `set_params`.
88pub struct Reverb {
89    /// The four parallel feedback comb filters.
90    combs: [CombFilter; 4],
91    /// The two serial allpass diffusion stages.
92    allpasses: [AllpassFilter; 2],
93}
94
95impl Default for Reverb {
96    /// Create a reverb with default room size and damping.
97    fn default() -> Self {
98        Self::new()
99    }
100}
101
102impl Reverb {
103    /// Construct a new reverb unit with pre-allocated delay buffers.
104    #[must_use]
105    pub fn new() -> Self {
106        Self {
107            combs: [
108                CombFilter::new(COMB_DELAYS[0]),
109                CombFilter::new(COMB_DELAYS[1]),
110                CombFilter::new(COMB_DELAYS[2]),
111                CombFilter::new(COMB_DELAYS[3]),
112            ],
113            allpasses: [
114                AllpassFilter::new(ALLPASS_DELAYS[0]),
115                AllpassFilter::new(ALLPASS_DELAYS[1]),
116            ],
117        }
118    }
119
120    /// Update room size and damping without reallocating delay buffers.
121    pub fn set_params(&mut self, size: f32, damping: f32) {
122        let feedback = 0.70 + size.clamp(0.0, 1.0) * 0.25; // 0.70 .. 0.95
123        let damp = damping.clamp(0.0, 0.999);
124        for c in &mut self.combs {
125            c.set_params(feedback, damp);
126        }
127    }
128
129    /// Process a mono sample.  Returns the wet+dry mix.
130    ///
131    /// `mix` – 0.0 (dry) .. 1.0 (full wet).
132    pub fn process(&mut self, input: f32, mix: f32) -> f32 {
133        if mix < 1e-4 {
134            return input;
135        }
136
137        // Sum 4 parallel combs (scale input to prevent overload)
138        let scaled = input * 0.015;
139        let mut wet = 0.0_f32;
140        for c in &mut self.combs {
141            wet += c.process(scaled);
142        }
143
144        // Two serial allpasses
145        for ap in &mut self.allpasses {
146            wet = ap.process(wet);
147        }
148
149        input * (1.0 - mix) + wet * mix * 5.0 // compensate comb scale-down
150    }
151}
