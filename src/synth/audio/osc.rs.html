1//! Oscillator, LFO, and supporting functions for pitch conversion.
2//!
3//! The `Oscillator` implements five waveform shapes with an integrated LFSR
4//! noise source clocked at the oscillator period boundary (SID-style behaviour).
5
6use crate::params::Waveform;
7use std::f32::consts::TAU;
8
9/// Single oscillator with LFSR-based noise (SID-style: LFSR clocked at osc frequency).
10pub struct Oscillator {
11    /// Current oscillator phase, normalised to 0.0 .. 1.0.
12    phase: f32,
13    /// 32-bit Galois LFSR state (feedback polynomial 0xB4BCD35C).
14    noise_lfsr: u32,
15    /// Most recent LFSR output, held between period boundaries.
16    last_noise: f32,
17}
18
19impl Default for Oscillator {
20    /// Create an oscillator with a non-zero LFSR seed to avoid the zero-lock state.
21    fn default() -> Self {
22        Self {
23            phase: 0.0,
24            noise_lfsr: 0xACE1_FEED,
25            last_noise: 0.0,
26        }
27    }
28}
29
30impl Oscillator {
31    /// Reset the phase accumulator to zero (useful for hard-sync effects).
32    #[allow(dead_code)]
33    pub fn reset(&mut self) {
34        self.phase = 0.0;
35    }
36
37    /// Returns the next sample in the range -1.0 .. 1.0.
38    ///
39    /// * `freq_hz`     – instantaneous frequency (already LFO-modulated if needed)
40    /// * `pulse_width` – 0.05 .. 0.95 (only relevant for Pulse / `PulseSaw`)
41    /// * `noise_mix`   – blend pure oscillator with raw LFSR noise
42    pub fn next_sample(
43        &mut self,
44        freq_hz: f32,
45        sample_rate: f32,
46        waveform: Waveform,
47        pulse_width: f32,
48        noise_mix: f32,
49    ) -> f32 {
50        let inc = freq_hz / sample_rate;
51        self.phase += inc;
52
53        // Phase wrap – tick LFSR on oscillator period boundary (SID behaviour)
54        if self.phase >= 1.0 {
55            self.phase -= 1.0;
56            self.last_noise = self.tick_lfsr();
57        }
58
59        let p = self.phase;
60
61        let osc = match waveform {
62            Waveform::Pulse => {
63                if p < pulse_width {
64                    1.0_f32
65                } else {
66                    -1.0_f32
67                }
68            }
69            Waveform::Sawtooth => 2.0 * p - 1.0,
70            Waveform::Triangle => {
71                if p < 0.5 {
72                    4.0 * p - 1.0
73                } else {
74                    3.0 - 4.0 * p
75                }
76            }
77            Waveform::Noise => self.last_noise,
78            Waveform::PulseSaw => {
79                let pulse = if p < pulse_width { 1.0_f32 } else { -1.0_f32 };
80                let saw = 2.0 * p - 1.0;
81                (pulse + saw) * 0.5
82            }
83        };
84
85        // Blend oscillator output with raw noise
86        if noise_mix > 0.001 {
87            osc * (1.0 - noise_mix) + self.last_noise * noise_mix
88        } else {
89            osc
90        }
91    }
92
93    /// Advance the 32-bit Galois LFSR by one step and return a sample in -1..1.
94    ///
95    /// Feedback polynomial: 0xB4BCD35C.  The LFSR is clocked once per oscillator
96    /// period (phase wrap), matching the SID chip's noise behaviour.
97    #[allow(clippy::cast_precision_loss)] // deliberate DSP normalisation; precision loss is acceptable
98    fn tick_lfsr(&mut self) -> f32 {
99        let bit = self.noise_lfsr & 1;
100        self.noise_lfsr >>= 1;
101        if bit != 0 {
102            self.noise_lfsr ^= 0xB4BC_D35C;
103        }
104        // Map u32 → -1..1 via signed reinterpretation (intentional wrapping cast)
105        self.noise_lfsr.cast_signed() as f32 / 2_147_483_648.0
106    }
107}

109/// Convert a MIDI note number to Hz (A4 = 69 = 440 Hz).
110#[inline]
111#[must_use]
112pub fn midi_to_hz(midi: u8) -> f32 {
113    440.0 * 2.0_f32.powf((f32::from(midi) - 69.0) / 12.0)
114}
115
116/// Apply detune in cents to a base frequency.
117#[inline]
118#[must_use]
119pub fn detune_hz(base_hz: f32, cents: f32) -> f32 {
120    base_hz * 2.0_f32.powf(cents / 1200.0)
121}

123/// Simple sine-wave LFO.
124pub struct Lfo {
125    /// Current LFO phase, normalised to 0.0 .. 1.0.
126    phase: f32,
127}
128
129impl Default for Lfo {
130    /// Create an LFO starting at phase zero.
131    fn default() -> Self {
132        Self { phase: 0.0 }
133    }
134}
135
136impl Lfo {
137    /// Advance the LFO by one sample and return a value in -1.0 .. 1.0.
138    pub fn next(&mut self, rate_hz: f32, sample_rate: f32) -> f32 {
139        self.phase += rate_hz / sample_rate;
140        if self.phase >= 1.0 {
141            self.phase -= 1.0;
142        }
143        (TAU * self.phase).sin()
144    }
145}

147#[cfg(test)]
148mod tests {
149    use super::*;
150
151    #[test]
152    fn osc_sawtooth_bounds() {
153        let mut osc = Oscillator::default();
154        for _ in 0..4410 {
155            let s = osc.next_sample(440.0, 44100.0, Waveform::Sawtooth, 0.5, 0.0);
156            assert!((-1.0..=1.0).contains(&s), "sawtooth out of bounds: {s}");
157        }
158    }
159
160    #[test]
161    #[allow(clippy::float_cmp)] // pulse wave output is exactly ±1.0 by construction
162    fn osc_pulse_bounds() {
163        let mut osc = Oscillator::default();
164        for _ in 0..4410 {
165            let s = osc.next_sample(440.0, 44100.0, Waveform::Pulse, 0.5, 0.0);
166            assert!(s == 1.0 || s == -1.0);
167        }
168    }
169
170    #[test]
171    fn midi_to_hz_a4() {
172        let hz = midi_to_hz(69);
173        assert!((hz - 440.0).abs() < 0.01);
174    }
